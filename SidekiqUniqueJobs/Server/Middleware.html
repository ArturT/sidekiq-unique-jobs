def call(worker_class, item, queue)
  @worker_class = worker_class
  @item         = item
  @queue        = queue
  return yield if unique_disabled?

  SidekiqUniqueJobs::Job.add_uniqueness(item)
  SidekiqUniqueJobs.with_context(logging_context(self.class, item)) do
    lock.execute do
      yield
    end
  end
end
