  1. item (required)

def lock_class
  @lock_class ||= begin
    locks.fetch(lock_type.to_sym) do
      raise UnknownLock, "No implementation for `lock: :#{lock_type}`"
    end
  end
end

def options
  @options ||= begin
    opts = default_worker_options.dup
    opts.merge!(worker_options) if sidekiq_worker_class?
    (opts || {}).stringify_keys
  end
end
