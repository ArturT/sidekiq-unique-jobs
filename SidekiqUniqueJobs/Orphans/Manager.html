def self.start
  with_logging_context do
    logger.info("Starting Reaper")
    task.add_observer(Observer.new)
    task.execute
    task
  end
end

def self.task
  @task ||= Concurrent::TimerTask.new(timer_task_options) do
    with_logging_context do
      redis do |conn|
        Orphans::Reaper.call(conn)
      end
    end
  end
end
